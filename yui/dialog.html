  • Async cancel, Async confirm
this.Dialog.confirm({
  title: "这个是标题",
  message: "这个是内容",
  beforeClose: (e) => {
    console.log("这个是会告诉你点击了哪个按钮", e);
    return new Promise((resolve, reject) => {
      console.log(reject);
      setTimeout(() => {
        resolve("请求结束");
      }, 3000);
    }).catch(err => {
      console.log("错误信息在这里展示", err);
    });
  }
});
